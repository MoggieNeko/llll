javascript:(()=>{const patterns=[/instagram\.com\/(reel|p)\//i,/tiktok\.com\//i,/youtube\.com\/(watch|shorts)/i,/youtu\.be\//i,/facebook\.com\/(watch|reel)/i,/(^|\.)x\.com\//i,/(^|\.)twitter\.com\//i];const urls=[...new Set([...document.querySelectorAll('a[href]')].map(a=>a.href).filter(h=>patterns.some(r=>r.test(h))))];const text=urls.join('\n');navigator.clipboard.writeText(text).then(()=>alert(`已複製 ${urls.length} 條網址到剪貼簿`)).catch(()=>prompt('請手動複製以下網址',text));})();
